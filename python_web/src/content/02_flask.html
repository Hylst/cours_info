Factory Pattern
from flask import Flask
from .models import db

def create_app(config_name='default'):
    app = Flask(__name__)
    app.config.from_object(f'config.Config')

    db.init_app(app)

    from .auth import auth_bp
    app.register_blueprint(auth_bp)

    return app
